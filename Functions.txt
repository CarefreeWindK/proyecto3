CREATE FUNCTION dbo.TCTMnombreToid (@nombre VARCHAR(32))
RETURNS INT
AS
BEGIN
    DECLARE @resultado INT;
    SELECT @resultado = A.Id
    FROM TCTM A
    WHERE A.Nombre = @nombre;
    RETURN @resultado;
END;

CREATE FUNCTION dbo.RNnombreToid (@nombre VARCHAR(32))
RETURNS INT
AS
BEGIN
    DECLARE @resultado INT;
    SELECT @resultado = A.Id
    FROM RN A
    WHERE A.Nombre = @nombre;
    RETURN @resultado;
END;
