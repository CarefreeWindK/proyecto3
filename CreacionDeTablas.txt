CREATE TABLE usuario
(
    Id INT IDENTITY(1,1) PRIMARY KEY
    ,Nombre VARCHAR(32)
	,Pin VARCHAR(32)
	,IdTH INT -- FK TH
    ,EsAdmin BIT
)
CREATE TABLE TipoDeTh
(
    Id INT IDENTITY(1,1) PRIMARY KEY
    ,Tipo VARCHAR(32)
	,Formato VARCHAR(32)
)


CREATE TABLE TH
(
    Id INT IDENTITY(1,1) PRIMARY KEY
    ,Nombre VARCHAR(32)
	,IdUsuario INT
	,IdCTM INT  -- FK CTM
	,IdCTA INT -- FK CTA
    ,IdTDTH INT -- FK TDTH
)


CREATE TABLE CTM
(
    Id INT IDENTITY (1, 1) NOT NULL PRIMARY KEY    
    , CodigoTCM INT NOT NULL -- Codigo de la tarjeta asociada
	, IdTCT INT NOT NULL -- FK TCT
    , IdTH INT NOT NULL -- FK TH
    , IdCT INT NOT NULL -- FK CT
	, IdCTA INT -- FK CTA
	, IdEC INT NOT NULL -- FK CTA
	, IdMS INT NOT NULL -- FK CTA
	, IdMov INT NOT NULL -- FK CTA
);



CREATE TABLE CTA
(

  Id INT IDENTITY (1, 1) NOT NULL PRIMARY KEY
  , CodigoTCA INT NOT NULL -- Codigo de la tarjeta asociada
  , IdTCM INT NOT NULL --FK CTM
  , IdTH INT NOT NULL -- FK TH
  , IdCT INT NOT NULL -- FK CT

);


CREATE TABLE CT
(
  Id INT IDENTITY (1, 1) NOT NULL PRIMARY KEY
  , EsMaestra BIT NOT NULL 
  , Codigo INT NOT NULL -- Codigo de la tarjeta asociada
);
CREATE TABLE TF
(
  Id INT IDENTITY (1, 1) NOT NULL PRIMARY KEY
  ,Nombre VARCHAR(32)
  ,FechaCaducacion VARCHAR(32)
  , CCV INT NOT NULL -- Codigo de la tarjeta asociada
  , Codigo INT NOT NULL -- Codigo de la tarjeta asociada
  , IdMI INT  --FK MI
  , IdMS INT  --FK MS
);


CREATE TABLE EC
(
  Id INT IDENTITY (1, 1) NOT NULL PRIMARY KEY
  ,Fecha VARCHAR(32)
  ,IdMov INT NOT NULL --FK Movimiento
);



CREATE TABLE MovimientoSospechoso
(
  Id INT IDENTITY (1, 1) NOT NULL PRIMARY KEY
  ,Monto INT NOT NULL
  ,Fecha VARCHAR(32)
  ,Descripcion VARCHAR(32)
  ,IdTF INT NOT NULL --FK TF
);




CREATE TABLE MotivoInvalidacion
(

    Id INT IDENTITY (1, 1) NOT NULL PRIMARY KEY
    , Nombre VARCHAR (64)

);
CREATE TABLE TMTI
(

  Id INT IDENTITY (1, 1) NOT NULL PRIMARY KEY
  , Nombre VARCHAR (64)  
  , Accion VARCHAR (16) --DEBITO O CREDITO

);
CREATE TABLE Movimiento
(

  Id INT IDENTITY (1, 1) NOT NULL PRIMARY KEY
  , Nombre VARCHAR (64)  
  , Fecha DATE
  , Monto MONEY
  , Descripcion VARCHAR (32)
  , Referencia VARCHAR (32)
  , IdTM INT --FK TM
  , IdTMTI INT --FK TMTI
  
);



CREATE TABLE RN

  Id INT IDENTITY (1, 1) NOT NULL PRIMARY KEY

CREATE TABLE TCTXRNQOP

CREATE TABLE TM
(

  Id INT IDENTITY (1, 1) NOT NULL PRIMARY KEY
  , Nombre VARCHAR (64)  
  , Accion VARCHAR (64)
  , Acumula_Operacion_ATM VARCHAR (8)
  , Acumula_Operacion_Ventana VARCHAR (8)

CREATE TABLE dbo.EventLog (
 Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY
 , LogDescription VARCHAR (2048) NOT NULL
 , PostIdUser INT NOT NULL
 , PostIP VARCHAR(64) NOT NULL
 , PostTime DATETIME NOT NULL
);
